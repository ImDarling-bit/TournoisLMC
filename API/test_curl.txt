
# ============================================================
#  Commandes curl de test - API Aux Claviers Citoyens
#  Organisation : par fichier REST + par CRUD
#
#  SYNTAXE PowerShell :
#    curl -Method  -> Invoke-WebRequest (natif PS, Body en guillemets simples)
#    curl.exe      -> vrai curl, utiliser --% pour bloquer l'interpretation PS
#                     ex: curl.exe --% -d "{\"email\":\"x\"}"
#
#  ETAPE 0 : s'assurer que l'utilisateur existe (register),
#            puis recuperer le token (login)
# ============================================================


# ------------------------------------------------------------
# ETAPE 0 : Creer l'utilisateur toto (si pas encore fait)
# ------------------------------------------------------------

curl.exe --% -X POST "http://localhost/Aux-Claviers-Citoyens/API/auth/register" -H "Content-Type: application/json" -d "{\"name\":\"Toto\",\"email\":\"toto@net.fr\",\"password\":\"toto\"}"

curl -Method POST -Uri "http://localhost/Aux-Claviers-Citoyens/API/auth/register" -ContentType "application/json" -Body '{"name":"Toto","email":"toto@net.fr","password":"toto"}'


# ------------------------------------------------------------
# ETAPE 1 : Login -> obtenir le token JWT
# ------------------------------------------------------------

curl.exe --% -X POST "http://localhost/Aux-Claviers-Citoyens/API/auth/token" -H "Content-Type: application/json" -d "{\"email\":\"toto@net.fr\",\"password\":\"toto\"}"

curl -Method POST -Uri "http://localhost/Aux-Claviers-Citoyens/API/auth/token" -ContentType "application/json" -Body '{"email":"toto@net.fr","password":"toto"}'


# Coller le token obtenu ici :
$token = "COLLER_LE_TOKEN_ICI"

$headers = @{
    Authorization  = "Bearer $token"
    'Content-Type' = 'application/json'
}

curl -Method POST -Uri "http://localhost/Aux-Claviers-Citoyens/API/auth/register" -ContentType "application/json" -Body '{"name":"Nouveau","email":"nouveau@test.fr","password":"pass123"}'

# --- CREATE : connexion (token) ---

curl.exe --% -X POST "http://localhost/Aux-Claviers-Citoyens/API/auth/token" -H "Content-Type: application/json" -d "{\"email\":\"toto@net.fr\",\"password\":\"toto\"}"

curl -Method POST -Uri "http://localhost/Aux-Claviers-Citoyens/API/auth/token" -ContentType "application/json" -Body '{"email":"toto@net.fr","password":"toto"}'

# --- READ : utilisateur connecte ---

curl.exe -X GET "http://localhost/Aux-Claviers-Citoyens/API/auth/me" -H "Authorization: Bearer $token"

curl -Method GET -Uri "http://localhost/Aux-Claviers-Citoyens/API/auth/me" -Headers @{ Authorization = "Bearer $token" }
